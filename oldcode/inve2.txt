export default function InventoryScreen() {
  const router = useRouter();
  const [searchQuery, setSearchQuery] = useState('');

  // Filtrar por búsqueda
  const filteredData = inventoryData.filter(item =>
    item.name.toLowerCase().includes(searchQuery.toLowerCase()) ||
    item.location.toLowerCase().includes(searchQuery.toLowerCase())
  );

  // Función simple para el botón REQUEST
  const handleRequest = (itemName: string) => {
    alert(`Solicitaste: ${itemName}`);
  };

  // Obtener color según estado
  const getStatusColor = (status: string) => {
    if (status === 'AVAILABLE') return '#4CAF50';
    if (status === 'IN USE') return '#FF9800';
    return '#F44336';
  };

  // Renderizar cada tarjeta
  const renderCard = ({ item }: { item: any }) => (
    <View style={styles.card}>
      <View style={styles.cardHeader}>
        <Text style={styles.itemName}>{item.name}</Text>
        <Text style={[styles.status, { color: getStatusColor(item.status) }]}>
          {item.status}
        </Text>
      </View>
      
      <Text style={styles.location}>{item.location}</Text>
      
      <TouchableOpacity 
        style={styles.requestButton}
        onPress={() => handleRequest(item.name)}
      >
        <Text style={styles.requestText}>REQUEST</Text>
      </TouchableOpacity>
    </View>
  );

  return (
    <SafeAreaView style={styles.container}>
      <StatusBar barStyle="dark-content" />
      
      {/* Header con botón back */}
      <View style={styles.header}>
        <TouchableOpacity onPress={() => router.back()}>
          <Ionicons name="arrow-back" size={24} color="#4A90E2" />
        </TouchableOpacity>
        <Text style={styles.title}>Total Resources</Text>
        <View style={{ width: 24 }} />
      </View>

      {/* Barra de búsqueda */}
      <View style={styles.searchContainer}>
        <Ionicons name="search-outline" size={20} color="#999" />
        <TextInput
          style={styles.searchInput}
          placeholder="Search inventory..."
          value={searchQuery}
          onChangeText={setSearchQuery}
        />
      </View>

      {/* Lista de recursos */}
      <FlatList
        data={filteredData}
        keyExtractor={(item) => item.id}
        renderItem={renderCard}
        contentContainerStyle={styles.list}
      />
    </SafeAreaView>
  );
}
