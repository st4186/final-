/**
 * Pantalla principal de la aplicación
 * Muestra los recursos en uso y permite navegar al inventario completo
 * 
 * @module HomeScreen
 */

import { useRouter } from 'expo-router';

    Dimensions,
    FlatList,
    SafeAreaView,
    StatusBar,
    StyleSheet,
    Text,
    TextInput,
    TouchableOpacity,
    View,
} from 'react-native';

// Dimensiones de la pantalla para ajustes responsivos
const { width, height } = Dimensions.get('window');

/**
 * Datos de ejemplo para los recursos
 * Simula la información que vendría de una API
 */
const resourcesData = [
  {
    id: '1',

    location: 'Building B- BL01',
    user: 'user@gmail.com',
    status: 'IN USE',
    statusColor: '#FF9800',

    statusColor: '#F44336',
    time: '9AM-10AM',
  },
];

/**
 * Componente principal de la pantalla Home
 */
export default function HomeScreen() {
  // Hook de navegación
  const router = useRouter();
  
  // Estado para el texto de búsqueda
  const [searchQuery, setSearchQuery] = useState('');
  
  // Estado para la pestaña activa en la navegación inferior
  const [activeTab, setActiveTab] = useState('home');

  /**
   * Filtra los recursos según el texto de búsqueda
   * Busca coincidencias en nombre y ubicación
   */
  const filteredResources = resourcesData.filter(item =>
    item.name.toLowerCase().includes(searchQuery.toLowerCase()) ||
    item.location.toLowerCase().includes(searchQuery.toLowerCase())
  );

  /**
   * Renderiza cada tarjeta de recurso
   * @param {Object} item - Objeto con la información del recurso
   */
  const renderResourceCard = ({ item }) => (
    <View style={styles.card}>
      {/* Encabezado de la tarjeta: nombre y estado */}
      <View style={styles.cardHeader}>
        <View style={styles.itemIcon}>
          <Ionicons name="cube-outline" size={18} color="#4A90E2" />
          <Text style={styles.itemName}>{item.name}</Text>
        </View>
        <View style={[styles.statusBadge, { backgroundColor: item.statusColor + '20' }]}>
          <Text style={[styles.statusText, { color: item.statusColor }]}>
            {item.status}
          </Text>
        </View>
      </View>
      
      {/* Ubicación del recurso */}
      <View style={styles.locationContainer}>
        <Ionicons name="location-outline" size={12} color="#888" />
        <Text style={styles.location}>{item.location}</Text>
      </View>
      
      {/* Horario (si aplica) */}
      {item.time && (
        <View style={styles.timeContainer}>
          <Ionicons name="time-outline" size={12} color="#888" />
          <Text style={styles.timeText}>TIME: {item.time}</Text>
        </View>
      )}
      
      {/* Usuario que utiliza el recurso */}
      <View style={styles.userContainer}>
        <Ionicons name="person-outline" size={12} color="#888" />
        <Text style={styles.userText}>{item.user}</Text>
      </View>
    </View>
  );

  return (
    <SafeAreaView style={styles.container}>
      {/* Barra de estado del sistema */}
      <StatusBar barStyle="dark-content" backgroundColor="#f8f9fa" />
      
      {/* Encabezado con logo e ícono de usuario */}
      <View style={styles.header}>
        <View>
          <Text style={styles.logo}>EDUTECH</Text>
          <Text style={styles.subtitle}>Resources available</Text>
        </View>
        <TouchableOpacity style={styles.userIcon} onPress={() => router.push('/profile')}>
          <Ionicons name="person-circle-outline" size={36} color="#4A90E2" />
        </TouchableOpacity>
      </View>
      
      {/* Barra de búsqueda */}
      <View style={styles.searchWrapper}>
        <View style={styles.searchContainer}>
          <View style={styles.searchBar}>
            <Ionicons name="search-outline" size={16} color="#999" />
            <TextInput
              style={styles.searchInput}
              placeholder="Check inventory"
              placeholderTextColor="#999"
              value={searchQuery}
              onChangeText={setSearchQuery}
            />
          </View>
          <TouchableOpacity style={styles.filterButton}>
            <Ionicons name="options-outline" size={16} color="#fff" />
          </TouchableOpacity>
        </View>
      </View>

      {/* Contenido principal */}
      <View style={styles.content}>
        {/* Botón para ver lista completa de recursos */}
        <TouchableOpacity 
          style={styles.seeListButton} 
          onPress={() => router.push('/inventory')}
        >
          <Text style={styles.seeListText}>SEE LIST</Text>
          <Ionicons name="arrow-forward" size={16} color="#4A90E2" />
        </TouchableOpacity>

        {/* Título de la sección de recursos en uso */}
        <View style={styles.sectionHeader}>
          <Text style={styles.sectionTitle}>IN USE AND LOCATION</Text>
          <Text style={styles.sectionCount}>{filteredResources.length} items</Text>
        </View>

        {/* Lista de tarjetas de recursos */}
        <FlatList
          data={filteredResources}
          renderItem={renderResourceCard}
          keyExtractor={(item) => item.id}
          showsVerticalScrollIndicator={false}
          contentContainerStyle={styles.cardsContainer}
          style={styles.cardsList}
        />
      </View>

      {/* Barra de navegación inferior */}
      <View style={styles.bottomNav}>
        {/* Botón Home */}
        <TouchableOpacity 
          style={styles.navItem} 
          onPress={() => setActiveTab('home')}
        >
          <Ionicons 
            name={activeTab === 'home' ? 'home' : 'home-outline'} 
            size={20} 
            color={activeTab === 'home' ? '#4A90E2' : '#999'} 
          />
          <Text style={[styles.navText, activeTab === 'home' && styles.navTextActive]}>
            Home
          </Text>
        </TouchableOpacity>
        
        {/* Botón Search - navega al inventario */}
        <TouchableOpacity 
          style={styles.navItem} 
          onPress={() => {
            setActiveTab('search');
            router.push('/inventory');
          }}
        >
          <Ionicons 
            name={activeTab === 'search' ? 'search' : 'search-outline'} 
            size={20} 
            color={activeTab === 'search' ? '#4A90E2' : '#999'} 
          />
          <Text style={[styles.navText, activeTab === 'search' && styles.navTextActive]}>
            Search
          </Text>
        </TouchableOpacity>
        
        {/* Botón Profile - navega al perfil */}
        <TouchableOpacity 
          style={styles.navItem} 
          onPress={() => router.push('/profile')}
        >
          <Ionicons 
            name={activeTab === 'profile' ? 'person' : 'person-outline'} 
            size={20} 
            color={activeTab === 'profile' ? '#4A90E2' : '#999'} 
          />
          <Text style={[styles.navText, activeTab === 'profile' && styles.navTextActive]}>
            Profile
          </Text>
        </TouchableOpacity>
      </View>
    </SafeAreaView>
  );
}
